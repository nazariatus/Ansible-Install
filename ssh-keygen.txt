# генеруємо приватний та публічний ключ на сервері Ansible
ssh-keygen
# називаємо його ansiblekey та два рази нажимаємо Enter після чого нам генерує приватний ключ ansiblekey та публічний ключ ansiblekey.pub

# копіюємо вміст публічного ключа ansiblekey.pub в файл "nano .ssh/authorized_keys" на сервері до якого ми хочемо логуватись з сервера Ansible
nano .ssh/authorized_keys

# права доступу до файлу authorized_keys мають бути -rw------- і якщо вони не такі то їх треба змінити
cd .ssh/
chmod 0600 authorized_keys

# щоб залогуватись на інший сервер потрібно виконати команду
ssh -i ansiblekey nazar@192.168.1.35
# тут -i означає що буде використано приватний ключ, ansiblekey це приватний ключ, nazar це користувач на сервері до якого конектяться, 192.168.1.35 це ip адрес сервера до якого конектяться

# при потребі можна зробити так, щоб "-i ansiblekey" писати було не потрібно. Для цього треба перейменуваати приватний ключ в id_rsa та помістити за шляхом cd .ssh/
cd ~
mv ansiblekey .ssh/id_rsa
# тепер можна підключатись командою ssh nazar@192.168.1.35
